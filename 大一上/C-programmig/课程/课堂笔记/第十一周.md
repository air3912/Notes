#### 显隐式转换
```c
int value1 = 10;
int value2 = 20;

int *p1 = &value1;           // p1 指向 value1
const int *a = &value2;      // a 指向 value2 (const 指针)

// 可以将非 const 指针赋值给 const 指针
a = p1;                      // OK, 现在 a 指向 value1, 非const 转 const

// 不能将 const 指针直接赋值给非 const 指针
// p1 = a;                   // 错误：舍弃限定符（const int* 到 int*）

p1 = (int*)a;                // 必须显式转型，现在 p1 和 a 都指向 value1

printf("value1 = %d\n", value1);  // 输出: 10
printf("value2 = %d\n", value2);  // 输出: 20
printf("*p1 = %d\n", *p1);        // 输出: 10
printf("*a = %d\n", *a);          // 输出: 10
```
